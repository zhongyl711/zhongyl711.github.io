# 通用过滤器

ynet-filter库，提供了账号、金额、日期Vue的全局过滤器

# API

# 账号

过滤器 说明
accountMask 同工具库accountUtil.mask方法
accountLast 同工具库accountUtil.last方法
accountFormat 同工具库accountUtil.format方法

# 金额

过滤器 说明
moneyFormat 同工具库moneyUtil.format方法
moneyParse 同工具库moneyUtil.parse方法
moneyToChinese 同工具库moneyUtil.toChinese方法

# 日期

过滤器 说明
dateFormat 同工具库dateUtil.format方法
timeFormat 同工具库dateUtil.formatTime方法

# 如何使用

# 1.项目package.json文件中导入@ynet/filter库

"dependencies": {     
    "@ynet/filter": "1.0.0"
}

# 2.引入并注册filter

// 1.在项目中按需引入即可
import { moneyFormat, accountMask, accountLast, accountFormat, dateFormat, timeFormat } from "@ynet/filter";
// 2. 上面导入的每个对象格式 {name: string, filter: function},可以直接传入filter函数里面，也可以像Vue.filter方式传入过滤器名字及函数
// 此处app对象参照文档 起步 》 核心 内容
export function filterSetting(app) {
    app.baseVue.filter(moneyFormat);
    app.baseVue.filter(accountMask);
    app.baseVue.filter(accountLast);
    app.baseVue.filter(accountFormat);
    app.baseVue.filter(dateFormat);
    app.baseVue.filter(timeFormat);

    // 静态图片地址替换
    app.baseVue.filter("imageUrlFormat", function(val) {
        if (!val) return;
        return val.replace("@@ebus1@@", ynet_process_cdn_url);
    });
}
