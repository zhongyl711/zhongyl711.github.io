# 编译逻辑分支插件

ynet-if-loader,提供按需编译加载内容功能，根据不同环境执行不同代码块

# 如何使用

# 1.项目package.json文件中导入@ynet/if-loader库

"dependencies": {
    "@ynet/if-loader": "1.0.0"
}

# 2.在build目录下配置文件中定义全局全局变量，因环境而异

1.在base中extendPlugin中添加插件

module.exports = {
    compilerOption: {
        entry: {
            main: "./main.js"
        }
    },
    extendPlugin: [
        [
            "@ynet/pxrem-loader",
            {
                //对应比例，750px设计稿
                remUnit: 37.5,
                //转化成rem时保留的小数位
                remPrecision: 8
            }
        ],
        "@ynet/if-loader"
    ]
};

2.在开发（dev）和 生产（prod）配置文件中 globalDefine 变量添加全局参数

module.exports = merge(baseConfig, {
    compilerOption: {
        devtool: "eval-source-map"
    },
    webpackServer: {
        enable: true,
        option: {
            port: 8801,
            proxy: {
                context: ["/MFP/financial/"],
                // target: "http://192.168.50.132:8801" //
                target: "https://ttt.ebanktest.com.cn:18106"
            }
        }
    },
    globalDefine: {
        ynet_process_loglevel: "'all'",
        ynet_process_request_type: "axios", // mock axios
        ynet_process_mode: "develop",
        // TODO: 生产环境需要配置
        ynet_process_cdn_url: JSON.stringify("https://801.ebanktest.com.cn:3802/mbank/static"), // 静态图片cdn
        ynet_process_pdf_url: JSON.stringify("https://ttt.ebanktest.com.cn:18106/MFP") // pdf
    },
    extendPlugin: ["@ynet/vconsole-plugin"]
});

3.在项目中应用，如在获取请求触发器中判断，如果ynet_process_request_type 是mock，则返回mock请求触发器，如果是axios，则返回axios触发器

注意： 以//#if 加判断条件（使用的全局变量前面必须加appConfig.globalDefine.）为起始，以//#endif结束

//按实际情况进行方法重写，实现特殊化需求
/** 自定义获取请求触发器 */
getRequestHandler(config) {
    //#if appConfig.globalDefine.ynet_process_request_type === "axios"
    // 设置请求上下文根
    config.url = "/MFP/financial/" + config.url;
    //#endif

    /**
         * 可根据不同的请求配置执行不同的请求处理逻辑
         * 例如：
         * a.do 使用mock
         * b.do 使用axios
         * ...
         */

    //#if appConfig.globalDefine.ynet_process_request_type === "mock"
    return new RequestByMock();
    //#endif

    //#if appConfig.globalDefine.ynet_process_request_type === "axios"
    // eslint-disable-next-line no-unreachable
    return new RequestByAxios();
    //#endif
}
