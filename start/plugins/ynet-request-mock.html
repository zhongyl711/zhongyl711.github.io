# mock请求处理

ynet-request-mock插件处理mock请求

# 如何使用

# 1.项目package.json文件中导入@ynet/request-mock库

"dependencies": {
	"@ynet/request-mock": "1.0.0"
}

# 2.在项目中应用

在请求配置文件中如config文件夹下创建了request.js

// 1.导入 RequesterConfig 抽象类 需要实现其中配置 内容配置参见下面
import { RequesterConfig } from "@ynet/core";
// 2.导入mock插件
import { RequestByMock } from "@ynet/request-mock";
// 3.导入mock数据文件，from路径为mock文件路径，内容如何配置参见下方
import mockList from "../mock/index";
// 4.关联至mocklist属性上
RequestByMock.mocklist = mockList || {};

RequesterConfig抽象类实现

export class SelfRequestConfig extends RequesterConfig {
		/** 自定义获取请求触发器 */
    getRequestHandler(config) {
        //#if appConfig.globalDefine.ynet_process_request_type === "axios"
        // 设置请求上下文根
        config.url = "/MFP/financial/" + config.url;
        // config.url = "/MFP/financetest/" + config.url;
        //#endif

        /**
         * 可根据不同的请求配置执行不同的请求处理逻辑
         * 例如：
         * a.do 使用mock
         * b.do 使用axios
         * ...
         */
				// 返回mock请求触发器，实际业务中可以配合编译逻辑分支插件 来返回不同的控制器
        return new RequestByMock();                
    }
		// mock数据一般不关心这里公共配置，除非有特殊需求配置mock
    /** 自定义请求数据转换方法 */
    async transformReqData(data, config) {
        // 设置http 请求头
        config.headers = {
            "Content-Type": "application/json",
            //设备类型
            devicetype: "wap",
            //随机数
            randomnum: idsUtil.random(47),
            //token
            sid: "",
            //设备id
            did: ""
        };
        let result = {
            header: {},
            body: { ...data }
        };
    }

    /** 请求结果分析 */
    analyResult(content, success, error) {
        // mock数据默认就是成功返回
        success(content);
        return;       
    }
		/** 默认的错误处理函数 */
    defaultErrorFunc(err) {        
    		alert(err?.message || "请求出现错误，请您稍后再试");
    }
}

mock数据配置，如在mock文件夹下创建index.js文件内容如下：

1.导出对象的key值为接口名，value为函数，函数返回值数据作为该接口的mock数据。

2.可以根据config中的请求数据判断返回不同的mock数据，也可以自己随机生成数据返回。
