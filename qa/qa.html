# Q：如何实现跳转回退后，页面不重载（keepalive）？

1.创建根据路由配置信息控制是否重载，template文件如下

<div id="ynet-content" class="main-content">
  <keep-alive>
  	<router-view v-if="$route.meta.keepAlive" />
  </keep-alive>
  <router-view v-if="!$route.meta.keepAlive" />
</div>

2.路由配置文件如下，根据实际业务配置keepAlive，true为缓存页面，false为重载页面

export let routes = [
    {
        path: "/",
        redirect: "/index/index"
    },
    {
        path: "/index/index",
        name: "index",
        component: HomeIndex,
        meta: {
            title: "理财",
            checkLogin: false,
            keepAlive: false
        }
    },
    {
        path: "/index/product-list",
        name: "product-list",
        component: () => import("../view/index/product-list.vue"),
        meta: {
            title: "理财产品",
            checkLogin: false,
            keepAlive: true
        }
    },
 }

# Q：如何实现跨页面传值？

1.params:

a.参数相当于挂载到一个变量上，当刷新页面时，参数会丢失

this.$app.router.push({ name: "redeem", params: {name:"小明"} });

b.防止参数丢失，可以增加配置项disableRouterParamSession为false

this.$app.router.push({ name: "redeem", params: {name:"小明"}, disableRouterParamSession: false });

获取参数方法

this.getRouterParam()

2.query：参数会拼接在浏览器地址后

this.$app.router.push({ path: "/detail/historical-net-worth", query:{name:"小明"} });

获取query对象

this.$route.query

# Q：如何判断返回到根目录？

1.当前地址hash和第一次进入页面hash值进行对比，一致的话，当前应用已到根目录

2.当前地址和配置的indexPath进行对比（创建应用new Application传入的参数），一致的话，当前应用已到根目录

# Q：如何实现在路由跳转时进行权限判断？

在路由配置项中添加onBefore函数，执行跳转权限判断

export let routerOption = {
    routes,//路由配置数组
    onExit: () => {
        if (inApp()) {
            //关闭当前webview 返回上级
            execJsapi("closeWebContainer");
        }
    },
    onBefore: (to, from, next) => {
      //这里进行一些跳转判断 可以拦截跳转，或者满足某些需求跳转至其它页面
    }
}

# Q：未找到实现类，缺失实现，请检查，接口ID：xxxx 如何解决？

1.提示接口ID：YNET_STORAGE_SESSION，需要注入会话缓存

2.提示接口ID：YNET_STORAGE_LOCAL，需要注入持久化缓存

3.提示接口ID：YNET_REQUESTER_CONFIG，需要注入请求管理实现类

import {
    IContainer,
    STORAGESESSIONINJECTTAGNAME,
    STORAGELOCALINJECTTAGNAME,
    REQUESTERCONFIGINJECTTAGNAME
} from "@ynet/core";
import { SessionStorage } from "@ynet/session-storage";
import { SessionWindow } from "@ynet/session-window";
import { LocalStorage } from "@ynet/local-storage";
import { SelfRequestConfig, requestSetting } from "./config/request";
//注入会话缓存实现类(降级方案)
IContainer.bind(STORAGESESSIONINJECTTAGNAME).to(SessionStorage.checkEnable() ? SessionStorage : SessionWindow);

//注入持久化缓存实现类
IContainer.bind(STORAGELOCALINJECTTAGNAME).to(LocalStorage);

//注入请求管理实现类
IContainer.bind(REQUESTERCONFIGINJECTTAGNAME).to(SelfRequestConfig);

请求配置类实现

import { RequesterConfig } from "@ynet/core";
export class SelfRequestConfig extends RequesterConfig {
    nativeInfo = undefined;
    //按实际情况进行方法重写，实现特殊化需求
    /** 自定义获取请求触发器 */
    getRequestHandler(config) {
        //#if appConfig.globalDefine.ynet_process_request_type === "axios"
        // 设置请求上下文根
        config.url = "/MFP/financial/" + config.url;
        // config.url = "/MFP/financetest/" + config.url;
        //#endif

        /**
         * 可根据不同的请求配置执行不同的请求处理逻辑
         * 例如：
         * a.do 使用mock
         * b.do 使用axios
         * ...
         */

        //#if appConfig.globalDefine.ynet_process_request_type === "mock"
        return new RequestByMock();
        //#endif

        //#if appConfig.globalDefine.ynet_process_request_type === "axios"
        // eslint-disable-next-line no-unreachable
        return new RequestByAxios();
        //#endif
    }

    /** 自定义请求数据转换方法 */
    async transformReqData(data, config) {
        // 设置http 请求头
        config.headers = {
            "Content-Type": "application/json",
            //设备类型
            devicetype: "wap",
            //随机数
            randomnum: idsUtil.random(47),
            //token
            sid: "",
            //设备id
            did: ""
        };
        let result = {
            header: {},
            body: { ...data }
        };
        return result;
    }

    /** 请求结果分析 */
    analyResult(content, success, error) {
        //#if appConfig.globalDefine.ynet_process_request_type === "mock"
        success(content);
        return;
        //#endif

        // eslint-disable-next-line no-unreachable
        if (content.errorCode === "0") {
            success(content.data);
        } else {
            error({
                code: content.errorCode,
                message: content.errorMsg,
                data: content.data
            });
        }
    }
    defaultErrorFunc(err) {
        alert(err?.message || "请求出现错误，请您稍后再试");
    }
}
